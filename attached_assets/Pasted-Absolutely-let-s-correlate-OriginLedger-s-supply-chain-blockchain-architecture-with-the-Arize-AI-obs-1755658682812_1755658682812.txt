import { trace, SpanStatusCode } from '@opentelemetry/api';

// Wrap an event recording function in tracing
export async function recordEventWithTrace(event: InsertEvent) {
  const tracer = trace.getTracer('OriginLedgerTracer');
  await tracer.startActiveSpan('SupplyChainEvent', { attributes: {
    assetId: event.assetId,
    participantId: event.participantId,
    action: event.action,
    location: event.location,
    timestamp: Date.now(),
  }}, async (span) => {
    try {
      // Record blockchain event
      await recordSupplyChainEvent(event);
      span.setStatus({ code: SpanStatusCode.OK });
    } catch (e) {
      span.setStatus({ code: SpanStatusCode.ERROR, message: String(e) });
      throw e;
    } finally {
      span.end();
    }
  });
}
